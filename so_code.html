Principe de fonctionnement

Le robot utilise un asservissement proportionnel :

  • Correction latérale (X) : Le robot calcule l'écart entre le centre de la cible et le centre de la caméra (160 px). Si l'objet est à droite, il tourne à droite.
  • Gestion de distance (Z) : On compare la largeur actuelle de l'objet avec une TARGET_SIZE idéale. Si l'objet est trop petit, le robot avance.
  • Mixage : On combine ces deux informations pour ajuster indépendamment la vitesse de la roue gauche et de la roue droite.
  • Anti-Friction : La constante MIN_PWM ajoute une "poussée" minimale pour éviter que le robot ne reste bloqué par la résistance de ses propres moteurs.

Code Source Arduino

#include <HUSKYLENS.h>
#include <Wire.h>

// Configuration des moteurs (Romeo BLE)
const int directionPin_M1 = 4; const int speedPin_M1 = 5;
const int speedPin_M2 = 6;     const int directionPin_M2 = 7;

HUSKYLENS huskylens;

// Paramètres de suivi
const int CENTER_X = 160; 
const int TARGET_SIZE = 65;
const int TOLERANCE_SIZE = 12;
const float Kp_Rotation = 1.1;
const float Kp_Distance = 0.9;
const int MIN_PWM = 45;

void setup() {
  Wire.begin();
  huskylens.begin(Wire);
  huskylens.writeAlgorithm(ALGORITHM_OBJECT_TRACKING);
}

void loop() {
  if (!huskylens.request() || !huskylens.available()) {
    carStop(); return;
  }
  
  while (huskylens.available()) {
    HUSKYLENSResult result = huskylens.read();
    if (result.command == COMMAND_RETURN_BLOCK && result.ID == 1) {
      
      // 1. Calcul des erreurs
      int errorX = result.xCenter - CENTER_X; 
      int errorSize = TARGET_SIZE - result.width; 
      
      // 2. Calcul des corrections
      int rotationCorrection = errorX * Kp_Rotation;
      int moveSpeed = (abs(errorSize) > TOLERANCE_SIZE) ? -errorSize * Kp_Distance : 0;

      // 3. Application aux moteurs
      int leftSpeed  = moveSpeed + rotationCorrection;
      int rightSpeed = moveSpeed - rotationCorrection;

      // Compensation de friction
      if (abs(leftSpeed) > 5) leftSpeed += (leftSpeed > 0) ? MIN_PWM : -MIN_PWM;
      if (abs(rightSpeed) > 5) rightSpeed += (rightSpeed > 0) ? MIN_PWM : -MIN_PWM;

      carDrive((leftSpeed >= 0), abs(leftSpeed), (rightSpeed >= 0), abs(rightSpeed));
    }
  }
}

void carDrive(bool lDir, int lSpd, bool rDir, int rSpd) {
  digitalWrite(directionPin_M2, !lDir); analogWrite(speedPin_M2, constrain(lSpd, 0, 150));
  digitalWrite(directionPin_M1, !rDir); analogWrite(speedPin_M1, constrain(rSpd, 0, 150));
}

void carStop() { analogWrite(speedPin_M1, 0); analogWrite(speedPin_M2, 0); }
