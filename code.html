Logique d'évitement d'obstacles & Code source Arduino

🧠 Logique d'Évitement du Robot

  • 1. Avancer par défaut : Le robot active ses moteurs arrière et progresse en ligne droite tant qu'aucun obstacle n'est détecté à courte portée.
  • 2. Détection critique : Si la distance mesurée face au robot par le capteur URM37 passe en dessous du seuil limite, les moteurs s'arrêtent immédiatement.
  • 3. Phase de Scan : Le servomoteur oriente le capteur ultrason à gauche (balayage), puis à droite pour enregistrer l'espace disponible de chaque côté.
  • 4. Prise de Décision : L'algorithme compare les deux distances de scan et ordonne un pivotement des roues vers la direction offrant le maximum d'espace libre avant de repartir en avant.

/*
 * Code de pilotage du Robot Cherokey 4WD avec évitement d'obstacles
 * Carte mère : RoMeo BLE V1.0 / Capteur : URM37 V3.2 / Servomoteur sur Pin 9
 */

#include <Servo.h>

// Attribution des pins pour la commande des moteurs
#define speedPin_M1 5        // Vitesse Moteur Gauche (PWM)
#define directionPin_M1 4    // Direction Moteur Gauche
#define speedPin_M2 6        // Vitesse Moteur Droit (PWM)
#define directionPin_M2 7    // Direction Moteur Droit

// Attribution des pins pour le capteur ultrason URM37
#define URPWM 3              // Broche Echo (Mesure de la largeur d'impulsion)
#define URTRIG 10            // Broche Trigger (Déclenchement du capteur)

Servo myservo;               // Création de l'objet pour piloter le servomoteur
int distanceSeuil = 25;      // Seuil de détection d'obstacle en centimètres

void setup() {
    // Configuration des broches moteurs en sortie
    pinMode(speedPin_M1, OUTPUT);
    pinMode(directionPin_M1, OUTPUT);
    pinMode(speedPin_M2, OUTPUT);
    pinMode(directionPin_M2, OUTPUT);
    
    // Configuration des broches du capteur URM37
    pinMode(URTRIG, OUTPUT);
    digitalWrite(URTRIG, HIGH); // Initialisation à l'état haut
    pinMode(URPWM, INPUT);
    
    myservo.attach(9);         // Associe le servomoteur à la pin numérique 9
    myservo.write(90);        // Aligne le capteur droit devant (90 degrés)
    
    Serial.begin(9600);
    delay(1000);
}

void loop() {
    int distanceFace = calculerDistance();
    
    if (distanceFace > 0 && distanceFace <= distanceSeuil) {
        arreterLeRobot();
        delay(500);
        analyserEnvironnement();
    } else {
        avancerEnLigneDroite();
    }
    delay(60);
}

// Fonction permettant de calculer la distance via le capteur URM37
int calculerDistance() {
    digitalWrite(URTRIG, LOW);
    digitalWrite(URTRIG, HIGH);
    
    unsigned long dureeImpulsion = pulseIn(URPWM, LOW, 25000);
    int distanceCm = dureeImpulsion / 50; // Conversion de la largeur d'impulsion en cm
    
    if (distanceCm == 0) return 999;   // Pas d'obstacle à portée
    return distanceCm;
}

void avancerEnLigneDroite() {
    digitalWrite(directionPin_M1, HIGH);
    analogWrite(speedPin_M1, 150); // Vitesse modérée
    digitalWrite(directionPin_M2, HIGH);
    analogWrite(speedPin_M2, 150);
}

void arreterLeRobot() {
    analogWrite(speedPin_M1, 0);
    analogWrite(speedPin_M2, 0);
}

void pivoterAGauche() {
    digitalWrite(directionPin_M1, LOW);  // Recule à gauche
    analogWrite(speedPin_M1, 180);
    digitalWrite(directionPin_M2, HIGH); // Avance à droite
    analogWrite(speedPin_M2, 180);
    delay(600);                         // Durée du pivot
    arreterLeRobot();
}

void pivoterADroite() {
    digitalWrite(directionPin_M1, HIGH); // Avance à gauche
    analogWrite(speedPin_M1, 180);
    digitalWrite(directionPin_M2, LOW);  // Recule à droite
    analogWrite(speedPin_M2, 180);
    delay(600);
    arreterLeRobot();
}

// Processus d'analyse à gauche et à droite pour choisir la meilleure voie
void analyserEnvironnement() {
    myservo.write(20);  // Regarde à droite
    delay(600);
    int distanceDroite = calculerDistance();
    
    myservo.write(160); // Regarde à gauche
    delay(800);
    int distanceGauche = calculerDistance();
    
    myservo.write(90);  // Repositionne au centre
    delay(500);
    
    if (distanceGauche > distanceDroite) {
        pivoterAGauche();
    } else {
        pivoterADroite();
    }
}
