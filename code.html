#include <Servo.h>

// --- CONFIGURATION MOTEURS ---
const int directionPin_M1 = 4; 
const int speedPin_M1 = 5;
const int speedPin_M2 = 6; 
const int directionPin_M2 = 7;
const int URTRIG = 10; 
const int URPWM = 3; 
Servo myservo;

// --- PARAMÈTRES ---
const int speedForward = 100;
const int speedTurn = 150; 
const int speedBack = 95; 
const int distanceSeuil = 25;

void setup() {
  Serial.begin(9600);
  pinMode(directionPin_M1, OUTPUT);
  pinMode(directionPin_M2, OUTPUT);
  pinMode(URTRIG, OUTPUT);
  pinMode(URPWM, INPUT);
  myservo.attach(9);
  myservo.write(90);
  delay(500);
}

void loop() {
  myservo.write(90);
  int distance = MeasureDistance();
  
  if (distance > distanceSeuil) {
    carDrive(HIGH, speedForward, HIGH, speedForward);
    delay(50);
  } else {
    carStop();
    delay(500);
    carDrive(LOW, speedBack, LOW, speedBack);
    delay(300);
    carStop();
    delay(400);
    
    myservo.write(30);
    delay(700);
    int dR = MeasureDistance();
    
    myservo.write(150);
    delay(700);
    int dL = MeasureDistance();
    
    myservo.write(90);
    delay(500);
    
    if (dL > dR) {
      carDrive(LOW, speedTurn, HIGH, speedTurn);
      delay(750);
    } else {
      carDrive(HIGH, speedTurn, LOW, speedTurn);
      delay(750);
    }
    carStop();
    delay(400);
  }
}

int MeasureDistance() {
  digitalWrite(URTRIG, LOW);
  digitalWrite(URTRIG, HIGH);
  unsigned long duration = pulseIn(URPWM, LOW, 50000);
  if (duration == 0) return 999;
  return duration / 50;
}

void carDrive(bool leftDir, int leftSpeed, bool rightDir, int rightSpeed) {
  digitalWrite(directionPin_M2, !leftDir);
  analogWrite(speedPin_M2, leftSpeed);
  digitalWrite(directionPin_M1, !rightDir);
  analogWrite(speedPin_M1, rightSpeed);
}

void carStop() {
  analogWrite(speedPin_M1, 0);
  analogWrite(speedPin_M2, 0);
}
