Code Source Arduino

Étude de l'algorithme décisionnel d'évitement d'obstacles en environnement C++ / Arduino.

Logique Globale de Décision

Le programme exécute une boucle continue. Le robot avance en ligne droite tout en mesurant constamment la distance face à lui. Dès qu'un obstacle franchit le seuil critique des 20 cm, le robot stoppe net sa course.

Le servomoteur oriente ensuite le capteur à ultrason à 30° (droite) puis à 150° (gauche). L'algorithme compare ces deux distances, choisit la voie la plus dégagée, effectue son pivot et force une avancée rapide pour s'extraire de l'impasse.

Code d'Évitement Complet

#include <Servo.h>

int speedPin_M1 = 5;     // Moteur Droit Vitesse
int directionPin_M1 = 4; // Moteur Droit Direction
int speedPin_M2 = 6;     // Moteur Gauche Vitesse
int directionPin_M2 = 7; // Moteur Gauche Direction

int URTRIG = 10;         // Trig ultrason
int URPWM = 3;           // Echo ultrason

Servo myservo;

void setup() {
  Serial.begin(9600);
  pinMode(directionPin_M1, OUTPUT);
  pinMode(directionPin_M2, OUTPUT);
  pinMode(URTRIG, OUTPUT);
  pinMode(URPWM, INPUT);
  myservo.attach(9);
  myservo.write(90); // Capteur centré au départ
}

int MeasureDistance() {
  digitalWrite(URTRIG, LOW);
  digitalWrite(URTRIG, HIGH);
  unsigned long duration = pulseIn(URPWM, LOW, 50000);
  if (duration == 0) return 999;
  return duration / 50;
}

void carAdvance(int leftSpeed, int rightSpeed) {
  digitalWrite(directionPin_M1, HIGH);
  digitalWrite(directionPin_M2, HIGH);
  analogWrite(speedPin_M1, rightSpeed);
  analogWrite(speedPin_M2, leftSpeed);
}

void carBack(int leftSpeed, int rightSpeed) {
  digitalWrite(directionPin_M1, LOW);
  digitalWrite(directionPin_M2, LOW);
  analogWrite(speedPin_M1, rightSpeed);
  analogWrite(speedPin_M2, leftSpeed);
}

void carTurnLeft(int leftSpeed, int rightSpeed) {
  digitalWrite(directionPin_M1, HIGH);
  digitalWrite(directionPin_M2, LOW);
  analogWrite(speedPin_M1, rightSpeed);
  analogWrite(speedPin_M2, leftSpeed);
}

void carTurnRight(int leftSpeed, int rightSpeed) {
  digitalWrite(directionPin_M1, LOW);
  digitalWrite(directionPin_M2, HIGH);
  analogWrite(speedPin_M1, rightSpeed);
  analogWrite(speedPin_M2, leftSpeed);
}

void carStop() {
  analogWrite(speedPin_M1, 0);
  analogWrite(speedPin_M2, 0);
}

void loop() {
  myservo.write(90);
  int distance = MeasureDistance();
  
  if (distance > 20) {
    carAdvance(150, 150);
    delay(50);
  } else {
    carStop();
    delay(400);
    
    // Scan Droite (30°)
    myservo.write(30);
    delay(600);
    int dR = MeasureDistance();
    
    // Scan Gauche (150°)
    myservo.write(150);
    delay(600);
    int dL = MeasureDistance();
    
    myservo.write(90);
    delay(400);
    
    // Prise de décision
    if (dL > dR) {
      carTurnLeft(200, 200);
      delay(450);
    } else {
      carTurnRight(200, 200);
      delay(450);
    }
    carStop();
    delay(200);
    
    // Forçage d'extraction
    carAdvance(150, 150);
    delay(600);
  }
}
