1. Objectifs & Explications

Le système assure une navigation autonome fluide via :

  • Détection : Algorithme "Line Tracking" HuskyLens.
  • Asservissement (P) : Correction proportionnelle (Kp=0.7).
  • Sécurité : Interception de virage 90° avec mémoire.

2. Algorithme (Code)

#include <HUSKYLENS.h>
#include <Wire.h>

const int directionPin_M1 = 4; const int speedPin_M1 = 5;
const int speedPin_M2 = 6; const int directionPin_M2 = 7;

HUSKYLENS huskylens;
const int CENTER_X = 160; const int BASE_SPEED = 90;
const float Kp = 0.7; int lastDirection = 1;
bool isLineCurrentlyLost = false; unsigned long timeLost = 0;

void setup() {
  Wire.begin(); pinMode(directionPin_M1, OUTPUT); pinMode(speedPin_M1, OUTPUT);
  pinMode(speedPin_M2, OUTPUT); pinMode(directionPin_M2, OUTPUT);
  while (!huskylens.begin(Wire));
  huskylens.writeAlgorithm(ALGORITHM_LINE_TRACKING);
}

void loop() {
  if (!huskylens.request() || !huskylens.available()) {
    if (!isLineCurrentlyLost) { timeLost = millis(); isLineCurrentlyLost = true; }
    if (millis() - timeLost > 80) { intercepterVirageSerre(); }
    else { carDrive(HIGH, BASE_SPEED, HIGH, BASE_SPEED); }
    return;
  }
  isLineCurrentlyLost = false;
  while (huskylens.available()) {
    HUSKYLENSResult result = huskylens.read();
    if (result.command == COMMAND_RETURN_ARROW) {
      int error = result.xOrigin - CENTER_X;
      if (abs(error) > 40) lastDirection = (error > 0) ? 1 : -1;
      int steerCorrection = error * Kp;
      carDrive(HIGH, BASE_SPEED + steerCorrection, HIGH, BASE_SPEED - steerCorrection);
    }
  }
}

void intercepterVirageSerre() {
  int pivotSpeed = 140;
  if (lastDirection == 1) carDrive(HIGH, pivotSpeed, LOW, pivotSpeed);
  else carDrive(LOW, pivotSpeed, HIGH, pivotSpeed);
}

void carDrive(bool lD, int lS, bool rD, int rS) {
  digitalWrite(directionPin_M2, !lD); analogWrite(speedPin_M2, lS);
  digitalWrite(directionPin_M1, !rD); analogWrite(speedPin_M1, rS);
}
